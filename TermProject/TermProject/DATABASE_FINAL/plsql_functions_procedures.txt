CREATE OR REPLACE FUNCTION "ARTIST_OR_DIRECTOR"(NAAM IN VARCHAR2)
RETURN NUMBER IS

	CHOICE NUMBER;
	ACOUNT NUMBER;
	DCOUNT NUMBER;
	
BEGIN
	-- routine body goes here, e.g.
	-- DBMS_OUTPUT.PUT_LINE('Navicat for Oracle');
	SELECT COUNT("aID") INTO ACOUNT
	FROM ARTIST
	WHERE UPPER("Name") LIKE UPPER(NAAM);
	
	SELECT COUNT("dID") INTO DCOUNT
	FROM DIRECTOR
	WHERE UPPER("Name") LIKE UPPER(NAAM);
	
	IF ACOUNT >= 1 THEN
			CHOICE := 1;
			
	ELSIF DCOUNT >= 1 THEN
			CHOICE := 0;
			
	ELSE 
			CHOICE := -1;
			
	END IF;
		
	RETURN CHOICE;
END;
